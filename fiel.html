import streamlit as st import time # Configuración de la página st.set_page_config( page_title="Consola del Pastor", page_icon="✝️", layout="wide" ) # CSS personalizado para el estilo oscuro st.markdown(""" """, unsafe_allow_html=True) # Inicializar estado if 'texto_actual' not in st.session_state: st.session_state.texto_actual = "Esperando al pastor..." # Crear dos columnas: una para controles y otra para visualización col1, col2 = st.columns([1, 2]) with col1: st.header("📝 Panel de Control") # Campo de texto para el pastor nuevo_texto = st.text_area( "Escribe el mensaje:", height=200, placeholder="Escribe aquí el mensaje para mostrar..." ) # Botón para actualizar if st.button("📢 Actualizar Pantalla", type="primary"): if nuevo_texto: st.session_state.texto_actual = nuevo_texto st.success("✅ Mensaje actualizado") else: st.warning("⚠️ Escribe algo primero") # Botón para limpiar if st.button("🗑️ Limpiar Pantalla"): st.session_state.texto_actual = "Esperando al pastor..." st.info("Pantalla limpiada") with col2: st.header("📺 Pantalla Principal") # Mostrar el texto grande st.markdown(f"""
✅ Conectado
{st.session_state.texto_actual}
""", unsafe_allow_html=True) # Auto-refresh cada 2 segundos time.sleep(0.1) st.rerun()
